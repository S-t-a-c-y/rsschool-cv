    function mergeArrays(arr1, arr2) {
        const res = [];
        if (arr1[0] > arr1[arr1.length - 1]) arr1 = arr1.reverse();
        if (arr2[0] > arr2[arr2.length - 1]) arr2 = arr2.reverse();

        let i = 0, j = 0, r = 0;

        while (i < arr1.length && j < arr2.length) {
            if (arr1[i] === arr2[j]) {
            i++;
            } else {
            res[r++] = arr1[i] < arr2[j] ? arr1[i++] : arr2[j++]
            }
        }

        return i < arr1.length ? res.concat(arr1.slice(i)) : res.concat(arr2.slice(j));
    }
